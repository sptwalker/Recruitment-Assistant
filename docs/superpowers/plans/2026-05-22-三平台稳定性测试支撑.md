> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 在不改采集业务逻辑的前提下，给三平台桥接 + 扩展加上"看门狗超时 + 学习模式误进检测 + 跨轮历史回归检测 + 自动产出 round 摘要"四件事，让人工逐平台跑稳定性测试时有可分析、可对比的数据。

**Tech Stack:** Python 3.10 (loguru, asyncio, pathlib, subprocess), Chrome Extension MV3 (vanilla JS), pytest（仅工具脚本）。

**Spec:** docs/superpowers/specs/2026-05-22-三平台稳定性测试支撑.md

```python
"""测试期看门狗 & 学习模式误进检测的纯逻辑。

bridge 在每条 emit 来到时调用 MisrouteDetector.on_event()，
得到需要 emit 的 misroute 事件 list；候选人 / 全局超时由
WatchdogState 持有时间戳，asyncio 任务定期巡检。
"""

from __future__ import annotations

from dataclasses import dataclass, field
from datetime import datetime
from typing import Any

CANDIDATE_TIMEOUT_SECONDS = 120
GLOBAL_TIMEOUT_SECONDS = 300

# 每个候选人的终态：到达后停止该 cid 的 watchdog
CANDIDATE_TERMINAL_EVENTS = frozenset({
    "resume_persist_confirmed",
    "candidate_skipped",
    "resume_request_success",
})


@dataclass
class CandidateState:
    persisted: bool = False
    auto_used: bool = False
    learned_used: bool = False
    last_event_at: datetime = field(default_factory=datetime.now)
    last_event_type: str = ""
    terminal: bool = False


class MisrouteDetector:
    """在 bridge 端按 emit 流维护 per-candidate state，
    收到 manual_download_learning_required 时返回需要 emit 的 misroute 事件。"""

    def __init__(self) -> None:
        self._state: dict[str, CandidateState] = {}

    @property
    def states(self) -> dict[str, CandidateState]:
        return self._state

    def on_event(self, event_type: str, candidate_id: str, payload: dict[str, Any]) -> list[dict[str, Any]]:
        out: list[dict[str, Any]] = []
        if not candidate_id:
            return out
        if event_type == "candidate_clicked":
            # 重置该 cid 的 state
            self._state[candidate_id] = CandidateState()
            return out

        st = self._state.setdefault(candidate_id, CandidateState())
        st.last_event_at = datetime.now()
        st.last_event_type = event_type

        if event_type == "auto_download_click_used":
            st.auto_used = True
        elif event_type == "learned_download_click_used":
            st.learned_used = True
        elif event_type == "resume_persist_confirmed":
            st.persisted = True
            st.terminal = True
        elif event_type in CANDIDATE_TERMINAL_EVENTS:
            st.terminal = True
        elif event_type == "manual_download_learning_required":
            sig = payload.get("candidate_signature", "")
            if st.persisted:
                out.append({
                    "kind": "A2",
                    "candidate_id": candidate_id,
                    "candidate_signature": sig,
                    "note": "已落库却又弹学习模式（重复触发）",
                })
            if st.learned_used:
                out.append({
                    "kind": "A3",
                    "candidate_id": candidate_id,
                    "candidate_signature": sig,
                    "note": "learnedClick 已用却又弹学习模式（学习成果未生效）",
                })
        return out


@dataclass
class TimedOutCandidate:
    candidate_id: str
    last_event_type: str
    elapsed_seconds: float


@dataclass
class WatchdogState:
    """看门狗状态：bridge 用 datetime 走，asyncio 任务定期 poll check_candidates() / check_global()。"""

    candidate_timeout: int = CANDIDATE_TIMEOUT_SECONDS
    global_timeout: int = GLOBAL_TIMEOUT_SECONDS
    global_last_event_at: datetime | None = None
    detector: MisrouteDetector = field(default_factory=MisrouteDetector)

    def on_event(self, event_type: str, candidate_id: str, payload: dict[str, Any]) -> list[dict[str, Any]]:
        # 心跳不算事件
        if event_type != "heartbeat":
            self.global_last_event_at = datetime.now()
        return self.detector.on_event(event_type, candidate_id, payload)

    def check_candidates(self, now: datetime | None = None) -> list[TimedOutCandidate]:
        now = now or datetime.now()
        timed_out: list[TimedOutCandidate] = []
        for cid, st in list(self.detector.states.items()):
            if st.terminal:
                continue
            elapsed = (now - st.last_event_at).total_seconds()
            if elapsed >= self.candidate_timeout:
                timed_out.append(TimedOutCandidate(
                    candidate_id=cid,
                    last_event_type=st.last_event_type,
                    elapsed_seconds=elapsed,
                ))
                # 标记为终态，避免重复触发
                st.terminal = True
        return timed_out

    def check_global(self, now: datetime | None = None) -> float | None:
        if not self.global_last_event_at:
            return None
        now = now or datetime.now()
        elapsed = (now - self.global_last_event_at).total_seconds()
        if elapsed >= self.global_timeout:
            return elapsed
        return None

    def reset(self) -> None:
        self.global_last_event_at = None
        self.detector = MisrouteDetector()
```

```python
    def _start_watchdog_loop(self) -> None:
        """在 ws_server 的 asyncio loop 上启动看门狗巡检任务。"""
        loop = self.ws_server._loop
        if loop is None:
            self._log("warning", "看门狗未启动：ws_server 事件循环未就绪")
            return
        if self._watchdog_task and not self._watchdog_task.done():
            return
        self._watchdog.global_last_event_at = datetime.now()

        async def _watch():
            try:
                while self.runtime_state.get("running"):
                    await asyncio.sleep(self._watchdog_poll_interval)
                    self._poll_watchdog()
            except asyncio.CancelledError:
                pass

        self._watchdog_task = asyncio.run_coroutine_threadsafe(_watch(), loop)
        self._write_event_log("watchdog_started", {
            "candidate_timeout_s": self._watchdog.candidate_timeout,
            "global_timeout_s": self._watchdog.global_timeout,
        })

    def _stop_watchdog_loop(self) -> None:
        task = self._watchdog_task
        if task is not None:
            try:
                task.cancel()
            except Exception:
                pass
        self._watchdog_task = None

    def _poll_watchdog(self) -> None:
        # 候选人级超时
        for to in self._watchdog.check_candidates():
            self._log("highlight", f"⚠️ 看门狗：候选人 {to.candidate_id} 已 {to.elapsed_seconds:.0f}s 无事件（最后事件 {to.last_event_type}），强制跳过")
            self._write_event_log("watchdog_candidate_timeout", {
                "candidate_id": to.candidate_id,
                "last_event_type": to.last_event_type,
                "elapsed_seconds": round(to.elapsed_seconds, 1),
            })
            command = {
                "type": "skip_current_candidate",
                "data": {"candidate_id": to.candidate_id, "reason": "watchdog"},
                "run_id": self.runtime_state.get("run_id", ""),
            }
            try:
                self.ws_server.send_command(command)
            except Exception as exc:
                self._log("warning", f"看门狗 skip 指令下发失败: {exc}")
        # 全局级超时
        idle = self._watchdog.check_global()
        if idle is not None:
            self._log("error", f"⚠️ 看门狗：全局 {idle:.0f}s 无事件，已强制终止采集")
            self._write_event_log("watchdog_global_idle_timeout", {"elapsed_seconds": round(idle, 1)})
            try:
                self._finish_crawl_task("failed", error_message="global_idle_timeout")
            except Exception as exc:
                self._log("warning", f"看门狗终止采集失败: {exc}")
            self.runtime_state["running"] = False
            self._stop_watchdog_loop()
```

```python
"""分析单轮测试 JSONL → 写 round 摘要 + 维护跨轮 history。

Usage:

from __future__ import annotations

import json
import sys
from datetime import datetime
from pathlib import Path
from typing import Any, Iterable

PROJECT_ROOT = Path(__file__).resolve().parents[1]
TEST_RUNS_DIR = PROJECT_ROOT / "logs" / "test_runs"
HISTORY_PATH = TEST_RUNS_DIR / "_persist_history.json"
COUNTER_PATH = TEST_RUNS_DIR / "_round_counter.json"

PLATFORM_FROM_DIR = {
    "zhilian_extension": "zhilian",
    "boss_extension": "boss",
    "qiancheng_extension": "qiancheng",
}

BUTTON_PATH_TYPES = (
    "auto_download_click_used",
    "learned_download_click_used",
    "manual_download_learning_success",
    "manual_download_learning_failed",

def detect_platform(jsonl_path: Path) -> str:
    for part in jsonl_path.parts:
        if part in PLATFORM_FROM_DIR:
            return PLATFORM_FROM_DIR[part]
    return "unknown"


def parse_jsonl(jsonl_path: Path) -> Iterable[dict[str, Any]]:
    with jsonl_path.open(encoding="utf-8") as f:
        for line in f:
            line = line.strip()
            if not line:
                continue
            try:
                yield json.loads(line)
            except json.JSONDecodeError:
                continue


def _load_json(path: Path, default: Any) -> Any:
    if not path.exists():
        return default
    try:
        return json.loads(path.read_text(encoding="utf-8"))
    except json.JSONDecodeError:
        return default


def _save_json(path: Path, data: Any) -> None:
    path.parent.mkdir(parents=True, exist_ok=True)
    path.write_text(json.dumps(data, ensure_ascii=False, indent=2), encoding="utf-8")


def increment_round_counter(counter_path: Path, platform: str) -> int:
    data = _load_json(counter_path, {})
    n = int(data.get(platform, 0)) + 1
    data[platform] = n
    _save_json(counter_path, data)
    return n


def update_history(history_path: Path, platform: str, persisted_sigs: list[str],
                   round_n: int, run_started: str) -> None:
    data = _load_json(history_path, {})
    bucket = data.setdefault(platform, {})
    for sig in persisted_sigs:
        if not sig:
            continue
        existing = bucket.get(sig)
        if existing is None:
            bucket[sig] = {
                "first_success_round": round_n,
                "first_success_at": run_started,
                "last_success_round": round_n,
                "last_success_at": run_started,
                "total_success": 1,
            }
        else:
            existing["last_success_round"] = round_n
            existing["last_success_at"] = run_started
            existing["total_success"] = int(existing.get("total_success", 0)) + 1
    _save_json(history_path, data)


def detect_history_regression(history_path: Path, platform: str,
                              learning_sigs: list[str]) -> list[str]:
    data = _load_json(history_path, {})
    bucket = data.get(platform, {})
    return [sig for sig in learning_sigs if sig and sig in bucket]


def bucket_button_path(events: list[dict]) -> dict[str, int]:
    counts: dict[str, int] = {k: 0 for k in BUTTON_PATH_TYPES}
    for e in events:
        et = e.get("payload", {}).get("type") or e.get("event")
        if et in counts:
            counts[et] += 1
    return counts


def aggregate_metrics(events: list[dict]) -> dict[str, Any]:
    persisted: list[str] = []
    learning: list[str] = []
    misroute: list[dict] = []
    watchdog_cand: list[dict] = []
    watchdog_global: list[dict] = []
    unrecognized: dict[str, list[str]] = {
        "resume_preview_not_found": [],
        "boss_svg_download_icon_not_found": [],
    }

    for e in events:
        # 内层 extension_event 的真实类型藏在 payload.type
        inner_type = e.get("payload", {}).get("type") or e.get("event") or ""
        data = e.get("payload", {}).get("data") or e.get("payload", {}) or {}
        if inner_type == "resume_persist_confirmed":
            sig = data.get("candidate_signature", "")
            if sig:
                persisted.append(sig)
        elif inner_type == "manual_download_learning_required":
            sig = data.get("candidate_signature", "")
            if sig:
                learning.append(sig)
        elif inner_type == "learning_misroute_detected" or e.get("event") == "learning_misroute_detected":
            misroute.append(e.get("payload", {}))
        elif e.get("event") == "watchdog_candidate_timeout":
            watchdog_cand.append(e.get("payload", {}))
        elif e.get("event") == "watchdog_global_idle_timeout":
            watchdog_global.append(e.get("payload", {}))
        elif inner_type in unrecognized:
            unrecognized[inner_type].append(data.get("candidate_signature", ""))

    return {
        "persisted_signatures": persisted,
        "learning_signatures": learning,
        "misroute_events": misroute,
        "watchdog_candidate_timeouts": watchdog_cand,
        "watchdog_global_timeouts": watchdog_global,
        "unrecognized_events": unrecognized,
        "button_paths": bucket_button_path(events),
    }

def previous_round_summary(test_runs_dir: Path, platform: str, current_round: int) -> Path | None:
    if current_round <= 1:

    pattern = f"round_{current_round - 1}_{platform}_*.md"

def diff_against_previous(prev_path: Path | None, current: dict[str, Any]) -> list[str]:
    """简易：从上一轮 .md 解析 button_paths / misroute / watchdog 数字，生成对比 bullet。"""
    if not prev_path or not prev_path.exists():
        return []
    text = prev_path.read_text(encoding="utf-8")
    lines: list[str] = []

    def parse_count(label: str) -> int | None:
        for line in text.splitlines():
            if line.startswith(f"| {label} |"):
                parts = [p.strip() for p in line.split("|")]
                try:
                    return int(parts[2])
                except (ValueError, IndexError):
                    return None
        return None

    for label in BUTTON_PATH_TYPES:
        prev_n = parse_count(label)
        cur_n = current["button_paths"].get(label, 0)
        if prev_n is None:
            continue
        delta = cur_n - prev_n
        if delta == 0:
            continue
        sign = "+" if delta > 0 else ""
        lines.append(f"- {label}: {prev_n} → {cur_n} ({sign}{delta})")

    return lines


def render_summary(jsonl_path: Path, platform: str, round_n: int,

    lines.append("")

    lines.append("## 未识别事件")
    lines.append("| 事件 | 次数 | 候选人样本 |")
    lines.append("|---|---|---|")
    for k, sigs in metrics["unrecognized_events"].items():
        sample = sigs[0] if sigs else "—"
        lines.append(f"| {k} | {len(sigs)} | {sample} |")
    lines.append("")

    if diff_lines:
        lines.append(f"## 与 Round {round_n - 1} 对比")
        lines.extend(diff_lines)
        lines.append("")

    return "\n".join(lines)


def main(argv: list[str]) -> int:
    if len(argv) < 2:
        print("usage: analyze_test_run.py <jsonl_path>", file=sys.stderr)
        return 2

    jsonl_path = Path(argv[1]).resolve()
    if not jsonl_path.exists():
        print(f"日志文件不存在：{jsonl_path}", file=sys.stderr)
        return 1

    platform = detect_platform(jsonl_path)
    events = list(parse_jsonl(jsonl_path))
    metrics = aggregate_metrics(events)
    run_id = find_run_id(events)
    run_started = find_run_started(events)

    TEST_RUNS_DIR.mkdir(parents=True, exist_ok=True)
    round_n = increment_round_counter(COUNTER_PATH, platform)
    update_history(HISTORY_PATH, platform, metrics["persisted_signatures"], round_n, run_started)
    regressions = detect_history_regression(HISTORY_PATH, platform, metrics["learning_signatures"])
    diff_lines = diff_against_previous(
        previous_round_summary(TEST_RUNS_DIR, platform, round_n),
        metrics,
    )

    summary = render_summary(jsonl_path, platform, round_n, metrics, regressions, diff_lines, run_id, run_started)
    out_name = f"round_{round_n}_{platform}_{datetime.now().strftime('%Y%m%d_%H%M%S')}.md"
    out_path = TEST_RUNS_DIR / out_name
    out_path.write_text(summary, encoding="utf-8")
    print(str(out_path))
    return 0


if __name__ == "__main__":
    raise SystemExit(main(sys.argv))
```

## 风险与缓解（plan 层面新增）

| 风险 | 缓解 |
|---|---|
| asyncio.run_coroutine_threadsafe 抛异常被吞 | `_start_watchdog_loop` 内 try/except 捕获 + warning 日志 |
| watchdog skip 指令到达扩展时候选人已自然推进 | 扩展端 `case "skip_current_candidate"` 仅清空 pending waiter，不主动调 skipCandidate；`watchdog_skip_received` event 让 bridge 看到双向确认 |
| analyze 脚本异常未被 bridge 看到 | bridge 用 `subprocess.Popen` + `stderr=DEVNULL` 故意丢；脚本异常发生在生成 .md 之前，下次手工跑命令可复现 |
| .gitignore 已忽略 logs/ 时 round 摘要也被忽略 | 这正是期望——测试制品不入 git |
| Windows 子进程编码 | `subprocess.Popen` 默认走 sys.executable + 系统 cp936，Python 内部统一 UTF-8 写入，无 stdout 被读所以编码问题无副作用 |
